The Name Keyboard
	
Keyboard Input
	If Onscreen Keyboard is selected
		backspace = backspace
		shift = shift
		enter = end
		E = "E"
		R = "R"
		<- = move index left
		-> = move index right
	Else
		backspace = cancel
		E = cancel
		R = forward
		enter = forward
		<- = move selection left
		-> = move selection right
		
Xbox Input
	A = press selected button
	B = backspace
	start = end
